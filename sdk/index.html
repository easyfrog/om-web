ooomap SDK for uni-app #

在使用uni-app开发小程序项目时, 同样使用微信小程序版本的SDK, 如果是开发web端h5, 还是推荐使用标准版本的SDK

您可以使用 ooomap-uniapp项目模板 ( Github, 码云Gitee ) 快速的进行小程序室内地图开发, 或参考其中的代码逻辑

TIP

hello world

kasle lksd

创建 ooomap 页面 #

在pages目录中创建地图页面

bash
pages/map/map.vue

将 ooomap.xcx.min.js 放置到 static 静态资源目录下 #

bash
static/ooomap.xcx.min.js

需要在App.vue中为page设置高度为100% #

css
<style>
	page {
		height: 100%;
	}
</style>

具体的使用方式如下 #

此示例代码使用的是vue3中的script setup语法模式. 此种方式相对代码简洁逻辑清晰, 用户也可以自行修改为vue2中的语法

map.vue

vue
<template>
	<view style="height: 100%;">
		<!-- 绘制3D地图所用的canvas -->
		<canvas type="webgl" id="ooomap" style="width: 100%; height: 100%;" disable-scroll @touchstart="touchStart" @touchmove="touchMove" @touchend="touchEnd"></canvas>
		<!-- 创建2d平面图像画布 -->
		<canvas type="2d" id="canvas2d" style="pointer-events: none;position: absolute;left:-1000px"></canvas>
	</view>
</template>

<script setup>
	import { ref, onMounted, onBeforeUnmount } from "vue"

	// 在uni-app中需要这样引入ooomap SDK
	var om = require("../../static/ooomap.xcx.min.js")

	// ooomap map 实例
	var map = null
	var canvas2d, canvas3d

	// 封装在微信小程序中得到domElement的方法
	function getElement(id, callback) {
		uni.createSelectorQuery().select(`#${id}`).node().exec(res => {
			var node = res[0].node

			if (callback) {
				callback(node)
			}
		})
	}

	onMounted(() => {
		getElement('ooomap', node => {
			canvas3d = node
			getElement('canvas2d', node => {
				canvas2d = node

				// 得到2个必需的 canvas 后, 创建 ooomap 对象
				createOMMap(canvas3d, canvas2d)
			})
		})
	})

	// 在离开页面时销毁地图对象
	onBeforeUnmount(() => {
		if (map) {
			map.dispose()
		}
	})

	function createOMMap(canvas3d, canvas2d) {
		// 创建 ooomap 地图对象
		map = new om.Map({

		   container: canvas3d,         // canvas type: webgl
		   canvas2d: canvas2d,        	// canvas type: 2d

		   // 在ooomap编辑器中生成，或由官方提供
		   verifyUrl: 'https://www.ooomap.com/ooomap-verify/check/50689bb01e0ac2a9d93bb18f1e1260f8',
		   appID: '87ae6a00e5ca4e33dd7e858a66b73475',

		})

		console.log('ooomap instance:', map)
	}

	// 必要的几个canvas3d事件
	function touchStart(e) {
        canvas3d.dispatchTouchEvent({
          ...e,
          type: 'touchstart'
        })
    }
    function touchMove(e) {
        canvas3d.dispatchTouchEvent({
          ...e,
          type: 'touchmove'
        })
    }
    function touchEnd(e) {
        canvas3d.dispatchTouchEvent({
          ...e,
          type: 'touchend'
        })
    }
</script>
